# Database Schema Setup Commands

# After setting up Supabase and getting your DATABASE_URL:

# 1. Set the environment variable temporarily
export DATABASE_URL="postgresql://postgres:[YOUR-PASSWORD]@db.[PROJECT-ID].supabase.co:5432/postgres"

# 2. Push the schema to Supabase
pnpm db:push

# 3. Verify in Supabase Dashboard that tables were created:
# - anime-web_user
# - anime-web_session  
# - anime-web_account
# - anime-web_verification
# - anime-web_anime_list
# - anime-web_favorites
# - anime-web_post

# 4. Test the connection
pnpm db:studio
